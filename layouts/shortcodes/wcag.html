"-" -}} {{- $url := "" -}} {{- if $slug -}} {{- $url = printf "https://www.w3.org/WAI/WCAG%s/Understanding/%s.html" $versie $slug -}} {{- else -}} {{- $url = printf "https://www.w3.org/WAI/WCAG%s/quickref/#qr-%s" $versie $anchor -}} {{- end -}} WCAG {{ $number }} — {{ $name }}
